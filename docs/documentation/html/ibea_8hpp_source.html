254 template <typename S = GASolution>
255 auto m_objective_bounds(std::vector<S> const &population) {
256 auto ub = std::numeric_limits<ObjectiveVector::value_type>::min();
257 auto lb = std::numeric_limits<ObjectiveVector::value_type>::max();
258 for (auto const &individual : population) {
259 for (auto const v : individual.objective_vector()) {
260 lb = std::min(lb, v);
261 ub = std::max(ub, v);
262 }
263 }
264 return std::make_pair(lb, ub);
265 }

277 template <typename S = GASolution>
278 auto m_scale_objective_vectors(std::vector<S> const &population, double const lb,
279 double const ub) {
280 std::vector<S> s = population;

282 auto ov = individual.objective_vector();

286 individual.set_objv(std::move(ov));

300 template <typename I, typename S = GASolution>
301 auto m_adaptive_factor(std::vector<S> const &population, I &&indicator) {
302 auto &&[lb, ub] = m_objective_bounds(population);
303 auto s = m_scale_objective_vectors(population, lb, ub);
304 auto c = std::numeric_limits<ObjectiveVector::value_type>::min();
305 for (std::size_t i = 0; i < s.size(); ++i) {
306 for (std::size_t j = 0; j < s.size(); ++j) {
307 if (i != j) {
308 c = std::max(c, std::abs(indicator(s[i], s[j])));
309 }
310 }
311 }
312 return c;
313 }

324 template <typename I, typename S = GASolution>
325 void m_fitness_assignment(std::vector<S> &population, double const k, I &&indicator) const {
326 for (std::size_t i = 0; i < population.size(); ++i) {
327 population[i].set_fitness(0);
328 for (std::size_t j = 0; j < population.size(); ++j) {
329 if (i != j) {
330 population[i].set_fitness(population[i].fitness() -
331 std::exp(-indicator(population[j], population[i]) / k));
332 }
333 }
334 }

349 template <typename I, typename S = GASolution>
350 void m_environmental_selection(std::vector<S> &population, double const k,
351 std::size_t population_max_size, I &&indicator) {
352 while (population.size() > population_max_size) {
353 std::size_t worst = 0;
354 for (std::size_t i = 0; i < population.size(); ++i) {
355 worst = population[i].fitness() < population[worst].fitness() ? i : worst;
356 }
357
358 std::swap(population[worst], population.back());
359 for (std::size_t i = 0; i < population.size() - 1; ++i) {
360 population[i].set_fitness(population[i].fitness() +
361 std::exp(-indicator(population.back(), population[i]) / k));
362 }
363 population.pop_back();
364 }
365 }
