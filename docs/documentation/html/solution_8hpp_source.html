124 DominanceType dominance(Solution const &solution) const {
125 assert(m_decision.size() == solution.m_decision.size());
126
127 DominanceType res = DominanceType::EQUAL;
128 for (decltype(m_objective.size()) i = 0; i < m_objective.size(); ++i) {
129 if (m_objective[i] < solution.m_objective[i]) {
130 if (res == DominanceType::DOMINATES) {
131 return DominanceType::INCOMPARABLE;
132 }
133 res = DominanceType::DOMINATED;
134 } else if (m_objective[i] > solution.m_objective[i]) {
135 if (res == DominanceType::DOMINATED) {
136 return DominanceType::INCOMPARABLE;
137 }
138 res = DominanceType::DOMINATES;
139 }
140 }
141 return res;
142 }

240 static std::vector<Solution> neighborhood_solutions(RMNKEval &eval, Solution const &original) {
241 std::vector<Solution> neighborhood;
242 neighborhood.reserve(original.decision_vector().size());
243
244 for (size_t i = 0; i < original.decision_vector().size(); ++i) {
245 auto decision_vector = original.decision_vector();
246 decision_vector[i] = !decision_vector[i];
247
248 neighborhood.emplace_back(eval, std::move(decision_vector));
249 }
250
251 for (size_t i = 0; i < original.decision_vector().size(); ++i) {
252 for (size_t j = i + 1; j < original.decision_vector().size(); ++j) {
253 if (original.decision_vector()[i] == original.decision_vector()[j]) {
254 continue;
255 }
256
257 auto decision_vector = original.decision_vector();
258 swap(decision_vector[i], decision_vector[j]);
259 neighborhood.emplace_back(eval, std::move(decision_vector));
260 }
261 }
