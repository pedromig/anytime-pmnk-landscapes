266 for (std::size_t i = 0; i < m_matting_pool_size; ++i) {

268 for (std::size_t j = 0; j < m_tournament_size - 1; ++j) {
269 auto other = distrib(m_rng);
270 best = population[other].fitness() > population[best].fitness() ? other : best;
271 }
